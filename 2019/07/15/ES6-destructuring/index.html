ES6-destructuring

Posted on 2019-07-15 | 次阅读

destructuring(解构[这个单词本身的意思是破坏])

目的：为了快速拿到某个值，简化代码量

解构对象

let obj = {
    name:'zhang',
    age:18
};
({name,age} = obj) //name : 'zhang'     age:18

//或者  效果是一样的
let {name,age} = obj;

看着变量的名称都是一样的，很难受，我们也可以这样做:

let {name:oName,age:oAge} = obj;
console.log(oName,oAge) //zhang 18

当变量没有同名的属性时，导致取不到值，最后返回undefined

let { bar, foo } = { foo: 'aaa', bar: 'bbb' };
foo // "aaa"
bar // "bbb"

let { baz } = { foo: 'aaa', bar: 'bbb' };
baz // undefined

默认赋值

let obj = {
    name: 'zhang',
    age: 18
};
let {name, age, sex = 'female'} = obj; 
//此时sex的值为'female'

//如果
let obj = {
    name: 'zhang',
    age: 18，
    sex: 'male'
};
let {name, age, sex = 'female'} = obj;
//此时sex的值就为'male'

类似的例子也有：

function sum(a = 10,b = 20) {
    return a + b;
}
sum(1) //返回的是21
//这边只是给    a   赋值了1，并没有给  b   赋值，则拿到 b 的默认值

结构数组（数组的本质是对象）

let arr = [1, 2, 3];
let {
    x,
    y,
    z
} = arr;
console.log(x, y, z)
//undefined undefined undefined

// 别忘了，数组的本质是对象
let {
    0: x,
    1: y,
    2: z
} = arr;
console.log(x, y, z) // 1 2 3

// 如果写成对象的形式觉得比较麻烦的话，其实'化妆'成arr认识的人就好啦
let [x, y, z] = arr;
console.log(x, y, z) // 1 2 3


//我们也可以得到该数组的长度
let {length} = arr;
console.log(length) // 3

价值体现

复杂数据里面需要分开操作的时候，此时 解构 就非常有用

一个对象或者数组里往往会有很多个值，但是这些值又不一定能够全部用上，所以我们可以挑选出我们需要的数据

let arr = [1, 2, 3, name:'zhang'];
let [,,, {name}] = arr;
//将前三个值放空，不要这么多


  let oJson = '{"a":"你", "b":"好", "c":"世","d":"界"}';
    let oRes = JSON.parse(oJson);
    // 获取到数据的两个方式
    // 1、原生js的方式
    let oA = oRes.a;
    let oB = oRes.b;
    let oC = oRes.c;
    let oD = oRes.d;
    // 2、利用解构的方式
    let {
        a,
        b,
        c,
        d
    } = oRes;

对象的解构赋值，可以很方便地讲现有对象的方法，赋值到某个变量

// 例一
let { log, sin, cos } = Math;

// 例二
const { log } = console;
log('hello') // hello
