ES6-箭头函数

Posted on 2019-07-15 | 次阅读

在讨论箭头函数之前，我们可以先了解一下，它具有哪些特点：

1、不用写function关键字

2、只能作为一个函数不能new，也不具有原型

3、参数不能重复命名

4、返回值可以不写return，但是有时需要配合{}

5、内部arguments this 由定义时外围最接近一层的非箭头函数的arguments 和 this 决定其值

1、ES5函数 VS ES6箭头函数

function sum(a, b) {
    return a + b;
}
//或者
var sum = function (a,b) {
    return a + b;
}

//ES6的箭头函数
let sum = (a, b) => {
    return  a + b;
}

2、箭头函数可以不写return

let sum = (a, b) => {
    a + b;
    //return undefined //这边如果不写return的话，这边相当于return了一个undefined
}
console.log(sum(10,20))//undefined

要想得到具体的值，我们可以这样写

let sum = (a, b) => a + b;
console.log(sum(10,20)) //30

输出为数组

let sum = (a, b) => [a, b];
console.log(sum(10,20)) //[10,20]

输出为对象

let sum = (a, b) => {a:a,b:b};
console.log(sum(10,20)) //error

系统会将大括号识别成函数体的大括号

let sum = (a,b) => {
    a:a,
    b:b //然而函数里面并没有这样的表达式，所以会报错
}

如果非要输出为对象，我们该怎么办？

let sum = (a, b) => ({a:a,b:b});
//只需将{}外面加一个()，变成表达式即可

let sum = (a, b) => ({
    a: a,
    b: b
});
console.log(sum(10, 20))  //{a: 10, b: 20}

3、参数不能重复命名

在ES6之前,下面的写法在控制台是不会报错的

function a(x,x) {}

在ES6里面，控制台报错，告诉我们重复的名字是不允许出现的

let a = (x,x) => {}
// Duplicate parameter name not allowed in this context

4、只能作为函数，不能new，也不具有原型

下面的例子，控制台告诉我们a不是一个构造函数

let a = (x, y) => {};
new a();
// a is not a constructor at demo.html:87

在控制台输入 a.prototype 返回的值是 undefined

5、内部arguments this 由定义时外围最接近一层的非箭头函数的arguments 和 this 决定其值

（1）arguments
function outer() {
    console.log(arguments)
    let sum = (a, b) => {
        console.log(arguments, a, b);//这里的arguments和outer函数里面arguments是一样的
    }
    sum(1, 2)
}
outer(4, 6, 7)

如果不用箭头函数，内部函数如果需要外部函数的arguments，只能先将arguments赋值给arg这个变量，然后内部函数再调用

function outer() {
    var arg = arguments;
    return function() {
        console.log(arg, arguments)
    }
}
outer(4, 6, 7)(1, 2)

如果是箭头函数，则不必这么麻烦

function outer() {
    return () => {
        console.log(arguments)
    }
}
outer(4, 6, 7)(1, 2)
（2）this

我们需要注意的，箭头函数里面的this在定义函数的时候就已经绑定了，而不是像非箭头函数在执行函数的时候绑定

let sum = () => {
        console.log(this);  // this 指向 window
    }
sum()

//此时外面没有非箭头函数，所以指向window
 var a = 'outerObj';
    let sum = () => {
        console.log(this.a); // outerObj
    }
    var obj = {
        a: 'innerObj',
        fn: sum
    }
    obj.fn();

//如果是正常的函数,谁调用this就指向谁
 var obj = {
        a: 'innerObj',
        fn: sum
    }

    function sum() {
        console.log(this.a); //innerObj
    }
    obj.fn()

下面这个例子里，sum函数为箭头函数，所以就先看它外层是否有非箭头函数，显然这里面是存在的fn函数；那么sum里面的this就是由fn里面的this决定，因此，this.a指向的就是obj里面的a

var a = 'outerObj';
let obj = {
    a: 'innerObj',
    fn() {
        let sum = () => {
            console.log(this.a) //innerObj
        }
        sum();
    }
}
obj.fn()

//在ES5中上面的写法如下：

var a = 'outerObj';
let obj = {
    a: 'innerObj',
    fn:function fn() {
        var _this = this;
        var sum = function sum() {
            console.log(_this.a) //innerObj
        }
        sum();
    }
}
obj.fn()

就算把sum函数return到外部也是不好使的,只需要记住箭头函数里面的this在定义函数的时候就已经绑定了

var a = 'outerObj';
    let obj = {
        a: 'innerObj',
        fn() {
            let sum = () => { 
                console.log(this) //innerObj
            }
            return sum;
        }
    }
    let outerSum = obj.fn()
    outerSum()
